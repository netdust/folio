import { z } from 'zod';

export const CommentKindSchema = z.enum([
  'comment', 'plan', 'result', 'error', 'approval', 'rejection', 'reply',
]);
export type CommentKind = z.infer<typeof CommentKindSchema>;

export const CommentVisibilitySchema = z.enum(['normal', 'internal']);
export type CommentVisibility = z.infer<typeof CommentVisibilitySchema>;

export const ResolvedMentionSchema = z.object({
  target: z.string().regex(/^(user|agent):.+$/),
  resolved: z.boolean(),
  resolvedId: z.string().optional(),
  resolvedType: z.enum(['agent', 'user']).optional(),
});
export type ResolvedMention = z.infer<typeof ResolvedMentionSchema>;

export const CommentFrontmatterSchema = z
  .object({
    author: z.string().regex(/^(user|agent):.+$/),
    kind: CommentKindSchema.default('comment'),
    visibility: CommentVisibilitySchema.default('normal'),
    mentions: z.array(ResolvedMentionSchema).default([]),
    edited_at: z.string().datetime().optional(),
    target_agent: z.string().optional(),
    run_id: z.string().uuid().optional(),
    deleted_at: z.string().datetime().optional(),
  })
  .refine(
    (d) => !(d.kind === 'approval' || d.kind === 'rejection') || !!d.target_agent,
    { message: 'target_agent is required when kind is approval or rejection' },
  )
  .refine(
    (d) => !d.target_agent || d.kind === 'approval' || d.kind === 'rejection',
    { message: 'target_agent is only valid when kind is approval or rejection' },
  );
export type CommentFrontmatter = z.infer<typeof CommentFrontmatterSchema>;

// apps/server/src/lib/mention-parser.ts
import type { ResolvedMention } from './comment-schema.ts';

const TOKEN_RE = /(?:^|\s)@([a-z][a-z0-9-]+)/g;
const KEYWORD_RE = /^(approved|rejected)[.,!]?$/i;

export interface ParseMentionsInput {
  body: string;
  workspaceAgents: { id: string; slug: string; allowedProjectIds: string[] | ['*'] }[];
  workspaceMembers: { id: string; email: string }[];
  currentProjectId: string;
}

export interface ApprovalIntent {
  kind: 'approval' | 'rejection';
  targetAgent: string;
  targetAgentId: string;
}

export interface ParseMentionsResult {
  mentions: ResolvedMention[];
  approvalIntent: ApprovalIntent | null;
}

export function parseMentions(input: ParseMentionsInput): ParseMentionsResult {
  const { body, workspaceAgents, workspaceMembers, currentProjectId } = input;
  const seen = new Set<string>();
  const mentions: ResolvedMention[] = [];

  // Pass 1: collect tokens + resolve.
  // We also remember each agent-mention's end-offset to find the keyword window later.
  type Hit = { slug: string; endOffset: number; agentId?: string };
  const hits: Hit[] = [];

  for (const match of body.matchAll(TOKEN_RE)) {
    const slug = match[1];
    const start = match.index ?? 0;
    const endOffset = start + match[0].length;
    const targetAsAgent = `agent:${slug}`;
    const targetAsUser  = `user:${slug}`;

    // Try agent resolve.
    const agent = workspaceAgents.find(a => a.slug === slug);
    if (agent) {
      const inAllowList = agent.allowedProjectIds[0] === '*' || agent.allowedProjectIds.includes(currentProjectId);
      if (inAllowList) {
        if (!seen.has(targetAsAgent)) {
          mentions.push({ target: targetAsAgent, resolved: true, resolvedId: agent.id, resolvedType: 'agent' });
          seen.add(targetAsAgent);
        }
        hits.push({ slug, endOffset, agentId: agent.id });
      } else {
        if (!seen.has(targetAsAgent)) {
          mentions.push({ target: targetAsAgent, resolved: false });
          seen.add(targetAsAgent);
        }
      }
      continue;
    }
    // Try member resolve.
    const matches = workspaceMembers.filter(m => m.email.split('@')[0] === slug);
    if (matches.length === 1) {
      const target = `user:${matches[0].id}`;
      if (!seen.has(target)) {
        mentions.push({ target, resolved: true, resolvedId: matches[0].id, resolvedType: 'user' });
        seen.add(target);
      }
    } else {
      if (!seen.has(targetAsUser)) {
        mentions.push({ target: targetAsUser, resolved: false });
        seen.add(targetAsUser);
      }
    }
  }

  // Pass 2: approval-intent detection. Scan the first two whitespace-delimited
  // tokens AFTER each resolved-agent mention. First match wins.
  let approvalIntent: ApprovalIntent | null = null;
  for (const hit of hits) {
    if (!hit.agentId) continue;
    const after = body.slice(hit.endOffset).trimStart();
    const next = after.split(/\s+/).slice(0, 2);
    for (const tok of next) {
      const m = tok.match(KEYWORD_RE);
      if (!m) continue;
      const kind = m[1].toLowerCase() === 'approved' ? 'approval' : 'rejection';
      approvalIntent = { kind, targetAgent: hit.slug, targetAgentId: hit.agentId };
      break;
    }
    if (approvalIntent) break;
  }

  return { mentions, approvalIntent };
}
```

import { cn } from './cn.ts';

interface TabItem<T extends string> {
  value: T;
  label: string;
  icon?: string;
  count?: number;
}

interface TabStripProps<T extends string> {
  value: T;
  items: TabItem<T>[];
  onChange: (next: T) => void;
}

export function TabStrip<T extends string>({ value, items, onChange }: TabStripProps<T>) {
  return (
    <div role="tablist" className="flex items-center gap-1 border-b border-border-light">
      {items.map((item) => (
        <button
          key={item.value}
          type="button"
          role="tab"
          aria-pressed={value === item.value}
          onClick={() => { if (value !== item.value) onChange(item.value); }}
          className={cn(
            'flex items-center gap-1.5 border-b-2 px-3 py-2 text-sm transition-colors',
            value === item.value
              ? 'border-primary text-fg'
              : 'border-transparent text-fg-2 hover:text-fg',
          )}
        >
          {item.icon ? <span aria-hidden>{item.icon}</span> : null}
          <span>{item.label}</span>
          {item.count != null && item.count > 0 ? (
            <span className="ml-0.5 rounded-full bg-shell px-1.5 text-[10px] text-fg-2">
              {item.count}
            </span>
          ) : null}
        </button>
      ))}
    </div>
  );
}

// apps/server/src/lib/reconciler.ts
import { db as defaultDb } from '../db/client.ts';
import { documents, projects } from '../db/schema.ts';
import { eq, and } from 'drizzle-orm';
import { emitEvent } from './events.ts';

export async function reconcileAllowLists(db = defaultDb, opts?: { onEvent?: (evt: any) => void }): Promise<void> {
  const agents = await db
    .select()
    .from(documents)
    .where(eq(documents.type, 'agent'));
  let touched = 0;
  for (const agent of agents) {
    const ids = (agent.frontmatter as any).projects as string[] | undefined;
    if (!ids || ids[0] === '*') continue;
    const liveProjects = await db
      .select({ id: projects.id })
      .from(projects)
      .where(/* projects.id in ids */);
    const liveSet = new Set(liveProjects.map(p => p.id));
    const filtered = ids.filter(id => liveSet.has(id));
    const removed = ids.filter(id => !liveSet.has(id));
    if (removed.length === 0) continue;
    await db.transaction(async (tx) => {
      await tx
        .update(documents)
        .set({ frontmatter: { ...(agent.frontmatter as any), projects: filtered } })
        .where(eq(documents.id, agent.id));
      await emitEvent(tx, {
        kind: 'agent.allow_list.reconciled',
        workspaceId: agent.workspaceId,
        documentId: agent.id,
        payload: { agent_id: agent.id, removed_project_ids: removed },
      });
    });
    touched++;
    opts?.onEvent?.({ kind: 'agent.allow_list.reconciled', removed });
  }
  if (touched > 0) console.log(`reconciler: scrubbed ${touched} agents`);
}
```
